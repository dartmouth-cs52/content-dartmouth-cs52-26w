function UserProfile() {
  const params = useParams();
  const [user, setUser] = useState({});

  useEffect(() => {
    const fetchData = async () => {
      const result = await axios(
        `https://api.com/users/${props.params.userId}`;,
      );
      setUser(result.data.user);
    };

    fetchData();
  }, []);
  
  return (
  	<h2>User: {user.name}</h2>
  )
}
