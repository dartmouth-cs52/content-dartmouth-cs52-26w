const nextState = produce(baseState, draftState => {
    draftState.push('newitem')
})

//OR

function nextStateProducer(state, argument) {
  return produce(state, draft => {
    draftState.push(argument)
  });
}

const nextState = nextStateProducer(baseState, 'newitem')
* just 1 function produce * takes the current state and a recipe * recipe is just a function that is given a draftState - a state you can modify a proxy object * and returns nextState - which is an optimized new object that may share structure
